### T1.9 — PostgreSQL 상세 초기화

#### postgresql.conf 튜닝 (컨테이너용 최적화)

`**config/postgres/postgresql.conf**`:

```ini
# 연결 설정
listen_addresses = '*'
max_connections = 100

# 메모리 설정 (컨테이너 512MB 기준)
shared_buffers = 128MB          # 전체 메모리의 25%
effective_cache_size = 384MB    # 전체 메모리의 75%
work_mem = 4MB
maintenance_work_mem = 64MB

# WAL 설정
wal_level = minimal
max_wal_size = 1GB
min_wal_size = 80MB
checkpoint_completion_target = 0.9

# 로그 설정
log_destination = 'stderr'
logging_collector = off
log_min_duration_statement = 1000   # 1초 이상 쿼리 로깅 (슬로우쿼리)
log_line_prefix = '%t [%p]: [%l-1] user=%u,db=%d '

# 타임존
timezone = 'Asia/Seoul'
log_timezone = 'Asia/Seoul'

# 한국어 정렬 지원
lc_messages = 'en_US.UTF-8'
lc_monetary = 'ko_KR.UTF-8'
lc_numeric = 'ko_KR.UTF-8'
lc_time = 'ko_KR.UTF-8'
```

#### DDL 스크립트

`**config/postgres/init.sql**`:

```sql
-- =============================================
-- AOMS 데이터베이스 초기화 스크립트
-- =============================================

SET client_encoding = 'UTF8';
SET timezone = 'Asia/Seoul';

-- ── 시스템 정보 ──────────────────────────────
CREATE TABLE systems (
    id              SERIAL PRIMARY KEY,
    system_name     VARCHAR(100) UNIQUE NOT NULL,
                    -- Prometheus label과 동일 (예: customer-experience)
    display_name    VARCHAR(200) NOT NULL,
                    -- 표시명 (예: 고객경험 시스템)
    description     TEXT,
    system_type     VARCHAR(50) NOT NULL,
                    -- 'web' | 'was' | 'db' | 'middleware' | 'pos'
    -- 이중화 구성 호스트 목록 (JSON)
    -- 예: [{"role":"was1","ip":"192.168.1.10","os":"linux"},...]
    hosts           JSONB NOT NULL DEFAULT '[]',
    status          VARCHAR(20) NOT NULL DEFAULT 'active',
                    -- 'active' | 'inactive'
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

COMMENT ON TABLE systems IS '모니터링 대상 시스템 목록';
COMMENT ON COLUMN systems.system_name IS 'Prometheus system_name 라벨 (이중화 시 논리 시스템명)';
COMMENT ON COLUMN systems.hosts IS '이중화 구성 호스트 목록 JSON';

-- ── 담당자 정보 ───────────────────────────────
CREATE TABLE contacts (
    id              SERIAL PRIMARY KEY,
    name            VARCHAR(100) NOT NULL,
    email           VARCHAR(200),
    teams_upn       VARCHAR(200),
                    -- Teams 알림용 UPN (예: user@company.com)
    webhook_url     TEXT,
                    -- 개인/팀 Webhook URL (선택)
    llm_api_key     VARCHAR(500),
                    -- 담당자별 LLM API key (AI 비용 분리 청구용)
                    -- NULL이면 시스템 기본 API key 사용
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

COMMENT ON COLUMN contacts.teams_upn IS 'Microsoft Teams UPN (알림 멘션용)';
COMMENT ON COLUMN contacts.llm_api_key IS '담당자별 LLM API key — NULL이면 .env 기본값 사용';

-- ── 시스템-담당자 매핑 ────────────────────────
CREATE TABLE system_contacts (
    id              SERIAL PRIMARY KEY,
    system_id       INTEGER NOT NULL REFERENCES systems(id) ON DELETE CASCADE,
    contact_id      INTEGER NOT NULL REFERENCES contacts(id) ON DELETE CASCADE,
    role            VARCHAR(50) NOT NULL DEFAULT 'primary',
                    -- 'primary' | 'secondary'
    notify_channels VARCHAR(200) NOT NULL DEFAULT 'teams',
                    -- 'teams' | 'teams,webhook' (콤마 구분)
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE (system_id, contact_id)
);

-- ── 알림 이력 ────────────────────────────────
CREATE TABLE alert_history (
    id                  SERIAL PRIMARY KEY,
    system_id           INTEGER REFERENCES systems(id) ON DELETE SET NULL,
    alert_type          VARCHAR(50)  NOT NULL,
                        -- 'metric' | 'log_analysis'
    severity            VARCHAR(20)  NOT NULL,
                        -- 'info' | 'warning' | 'critical'
    title               VARCHAR(500) NOT NULL,
    description         TEXT,
    -- 메트릭 알림 전용
    metric_name         VARCHAR(100),
    metric_value        FLOAT,
    instance_role       VARCHAR(100),
                        -- 어느 서버에서 발생했는지 (이중화 구분)
    host                VARCHAR(200),
    -- 발송 정보
    notified_contacts   JSONB,
                        -- 발송된 담당자 목록 [{name, channel}]
    teams_message_id    VARCHAR(500),
                        -- Teams 메시지 추적용 (선택)
    -- 확인 처리
    acknowledged        BOOLEAN NOT NULL DEFAULT FALSE,
    acknowledged_at     TIMESTAMPTZ,
    acknowledged_by     VARCHAR(100),
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

COMMENT ON COLUMN alert_history.instance_role IS '이중화 시스템에서 알림이 발생한 서버 역할 (was1, web2 등)';

-- ── LLM 분석 결과 이력 ───────────────────────
CREATE TABLE log_analysis_history (
    id              SERIAL PRIMARY KEY,
    system_id       INTEGER REFERENCES systems(id) ON DELETE SET NULL,
    instance_role   VARCHAR(100),
                    -- 어느 서버의 로그인지 (이중화 구분)
    log_content     TEXT NOT NULL,    -- 원본 로그 샘플 (최대 2000자)
    analysis_result TEXT NOT NULL,    -- LLM 전체 응답
    severity        VARCHAR(20) NOT NULL,
    root_cause      TEXT,
    recommendation  TEXT,
    model_used      VARCHAR(100),
    processing_time FLOAT,            -- LLM 응답 소요 시간 (초)
    alert_sent      BOOLEAN NOT NULL DEFAULT FALSE,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ── 알림 쿨다운 ──────────────────────────────
CREATE TABLE alert_cooldown (
    id              SERIAL PRIMARY KEY,
    system_id       INTEGER NOT NULL REFERENCES systems(id) ON DELETE CASCADE,
    alert_key       VARCHAR(500) NOT NULL,
                    -- '{system_name}:{instance_role}:{alertname}:{severity}'
    last_sent_at    TIMESTAMPTZ NOT NULL,
    UNIQUE (system_id, alert_key)
);

-- ── 피드백 이력 (Phase 4b: 벡터 DB 확장) ──────
CREATE TABLE alert_feedback (
    id                  BIGSERIAL PRIMARY KEY,
    alert_id            INTEGER REFERENCES alert_history(id),
    system_id           INTEGER REFERENCES systems(id),
    instance_role       VARCHAR(50),
    error_type          VARCHAR(100) NOT NULL,
    solution            TEXT NOT NULL,
    resolver            VARCHAR(100),
    resolution_time_min INTEGER,
    is_verified         BOOLEAN DEFAULT FALSE,
    qdrant_point_id     UUID,          -- Qdrant에 저장된 포인트 ID 추적
    saved_at            TIMESTAMP DEFAULT NOW(),
    updated_at          TIMESTAMP DEFAULT NOW()
);
COMMENT ON TABLE alert_feedback IS '담당자 오류 해결 피드백 (n8n Webhook 수신, Qdrant 벡터 연동)';
COMMENT ON COLUMN alert_feedback.qdrant_point_id IS 'Qdrant log_incidents 컬렉션의 포인트 UUID (피드백 등록 시 resolution payload 업데이트)';

-- ── 이상 감지 이력 (Phase 4b: 유사도 분석 결과) ─
CREATE TABLE anomaly_detections (
    id                  BIGSERIAL PRIMARY KEY,
    system_id           INTEGER REFERENCES systems(id),
    instance_role       VARCHAR(50),
    anomaly_type        VARCHAR(20) NOT NULL,
                        -- 'new' | 'recurring' | 'related' | 'duplicate'
    similarity_score    FLOAT,
    similar_point_id    UUID,                  -- Qdrant 유사 포인트 ID
    severity            VARCHAR(20),
    alert_sent          BOOLEAN DEFAULT FALSE,
    created_at          TIMESTAMP DEFAULT NOW()
);
COMMENT ON TABLE anomaly_detections IS '벡터 유사도 기반 이상 감지 결과 이력';
COMMENT ON COLUMN anomaly_detections.anomaly_type IS 'new(<0.75) | related(0.75~0.85) | recurring(0.85~0.95) | duplicate(≥0.95)';

-- ── 인덱스 ───────────────────────────────────
CREATE INDEX idx_alert_history_system     ON alert_history(system_id, created_at DESC);
CREATE INDEX idx_alert_history_created    ON alert_history(created_at DESC);
CREATE INDEX idx_alert_history_severity   ON alert_history(severity, created_at DESC);
CREATE INDEX idx_log_analysis_system      ON log_analysis_history(system_id, created_at DESC);
CREATE INDEX idx_log_analysis_created     ON log_analysis_history(created_at DESC);
CREATE INDEX idx_cooldown_lookup          ON alert_cooldown(system_id, alert_key);
CREATE INDEX idx_feedback_system      ON alert_feedback(system_id, saved_at DESC);
CREATE INDEX idx_anomaly_system       ON anomaly_detections(system_id, created_at DESC);
CREATE INDEX idx_anomaly_type         ON anomaly_detections(anomaly_type, created_at DESC);

-- ── 자동 updated_at 함수 ─────────────────────
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_systems_updated_at
    BEFORE UPDATE ON systems
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER trg_contacts_updated_at
    BEFORE UPDATE ON contacts
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();
```

### T1.11 — 파일럿 에이전트 설치 (자동 설치 스크립트)

#### Linux (RedHat 8.9) 자동 설치 스크립트

`**agents/linux/install-agents.sh**`:

```bash
#!/bin/bash
# ============================================================
# AOMS Linux 에이전트 자동 설치 스크립트 (폐쇄망용)
# 대상 OS: RedHat 8.9 / CentOS 8 (x86_64)
#
# 사용법:
#   ./install-agents.sh \
#     --system-name <논리시스템명> \
#     --instance-role <서버역할번호> \
#     --host <호스트명> \
#     --monitoring-server <모니터링서버IP> \
#     --install-dir <설치경로> \
#     --log-path <수집할로그경로패턴> \
#     [--type all|node|promtail|jmx] \
#     [--jmx-port <포트번호>]
#
# 예시:
#   ./install-agents.sh \
#     --system-name customer-experience \
#     --instance-role was1 \
#     --host cx-was01 \
#     --monitoring-server 192.168.10.5 \
#     --install-dir /opt/aoms-agents \
#     --log-path "/app/logs/*.log" \
#     --type all \
#     --jmx-port 9404
# ============================================================
set -euo pipefail

# ── 기본값 설정 ──────────────────────────────────────────────
INSTALL_DIR="/opt/aoms-agents"
AGENT_TYPE="all"
JMX_PORT=9404
NODE_EXPORTER_PORT=9100
PROMTAIL_PORT=9080
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

# ── 색상 출력 ────────────────────────────────────────────────
RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[1;33m'; NC='\033[0m'
info()    { echo -e "${GREEN}[INFO]${NC} $*"; }
warn()    { echo -e "${YELLOW}[WARN]${NC} $*"; }
error()   { echo -e "${RED}[ERROR]${NC} $*"; exit 1; }

# ── 파라미터 파싱 ────────────────────────────────────────────
while [[ $# -gt 0 ]]; do
    case "$1" in
        --system-name)      SYSTEM_NAME="$2";       shift 2 ;;
        --instance-role)    INSTANCE_ROLE="$2";     shift 2 ;;
        --host)             HOST_NAME="$2";          shift 2 ;;
        --monitoring-server) MONITORING_SERVER="$2"; shift 2 ;;
        --install-dir)      INSTALL_DIR="$2";        shift 2 ;;
        --log-path)         LOG_PATH="$2";           shift 2 ;;
        --type)             AGENT_TYPE="$2";         shift 2 ;;
        --jmx-port)         JMX_PORT="$2";           shift 2 ;;
        *) error "알 수 없는 옵션: $1" ;;
    esac
done

# ── 필수 파라미터 검증 ────────────────────────────────────────
[[ -z "${SYSTEM_NAME:-}"      ]] && error "--system-name 필수"
[[ -z "${INSTANCE_ROLE:-}"    ]] && error "--instance-role 필수"
[[ -z "${HOST_NAME:-}"        ]] && error "--host 필수"
[[ -z "${MONITORING_SERVER:-}"] && error "--monitoring-server 필수"
[[ -z "${LOG_PATH:-}"         ]] && error "--log-path 필수"

info "=== AOMS 에이전트 설치 시작 ==="
info "시스템명    : $SYSTEM_NAME"
info "서버 역할   : $INSTANCE_ROLE"
info "호스트명    : $HOST_NAME"
info "모니터링 서버: $MONITORING_SERVER"
info "설치 경로   : $INSTALL_DIR"
info "로그 경로   : $LOG_PATH"
info "에이전트 유형: $AGENT_TYPE"

# ── 설치 경로 생성 ────────────────────────────────────────────
install_node_exporter() {
    info "--- node_exporter 설치 ---"

    local AGENT_DIR="$INSTALL_DIR/node_exporter"
    local BIN_SRC="$SCRIPT_DIR/node_exporter-1.10.2.linux-amd64.tar.gz"
    local SERVICE_FILE="/etc/systemd/system/node_exporter.service"

    [[ -f "$BIN_SRC" ]] || error "node_exporter 바이너리 없음: $BIN_SRC"

    # 설치 디렉토리 생성
    sudo mkdir -p "$AGENT_DIR"

    # 압축 해제 및 바이너리 복사
    tar xzf "$BIN_SRC" -C /tmp/
    sudo cp /tmp/node_exporter-*/node_exporter "$AGENT_DIR/"
    sudo chmod +x "$AGENT_DIR/node_exporter"
    rm -rf /tmp/node_exporter-*/

    # node_exporter 전용 사용자 생성 (없으면)
    id node_exporter &>/dev/null || sudo useradd -r -s /bin/false node_exporter
    sudo chown node_exporter:node_exporter "$AGENT_DIR/node_exporter"

    # Systemd 서비스 등록
    sudo tee "$SERVICE_FILE" > /dev/null << EOF
[Unit]
Description=AOMS Node Exporter
Documentation=https://github.com/prometheus/node_exporter
After=network.target

[Service]
User=node_exporter
Group=node_exporter
Type=simple
ExecStart=$AGENT_DIR/node_exporter \
  --web.listen-address=:$NODE_EXPORTER_PORT \
  --collector.systemd \
  --collector.processes
Restart=on-failure
RestartSec=5s

[Install]
WantedBy=multi-user.target
EOF

    sudo systemctl daemon-reload
    sudo systemctl enable --now node_exporter

    # 방화벽 포트 허용
    sudo firewall-cmd --permanent --add-port=${NODE_EXPORTER_PORT}/tcp 2>/dev/null || true
    sudo firewall-cmd --reload 2>/dev/null || true

    # 검증
    sleep 2
    curl -sf "http://localhost:${NODE_EXPORTER_PORT}/metrics" | head -3 \
        && info "node_exporter 정상 기동 (포트 ${NODE_EXPORTER_PORT})" \
        || error "node_exporter 기동 실패"
}

install_promtail() {
    info "--- promtail 설치 ---"

    local AGENT_DIR="$INSTALL_DIR/promtail"
    local BIN_SRC="$SCRIPT_DIR/promtail-linux-amd64.zip"
    local CONFIG_FILE="$AGENT_DIR/promtail.yml"
    local POSITIONS_FILE="$AGENT_DIR/positions.yaml"
    local SERVICE_FILE="/etc/systemd/system/promtail.service"

    [[ -f "$BIN_SRC" ]] || error "promtail 바이너리 없음: $BIN_SRC"

    sudo mkdir -p "$AGENT_DIR"

    # 압축 해제 및 설치
    unzip -o "$BIN_SRC" -d /tmp/promtail_tmp/
    sudo cp /tmp/promtail_tmp/promtail-linux-amd64 "$AGENT_DIR/promtail"
    sudo chmod +x "$AGENT_DIR/promtail"
    rm -rf /tmp/promtail_tmp/

    # 전용 사용자 생성
    id promtail &>/dev/null || sudo useradd -r -s /bin/false promtail
    # 로그 파일 읽기 권한을 위해 adm/root 그룹에 추가
    sudo usermod -aG adm promtail 2>/dev/null || true

    # ─── promtail 설정 파일 생성 (방법3+4 조합) ───
    # 포괄적 멀티라인 + 빈줄 제거 + 에러 키워드 필터 + 레벨 라벨 추출
    sudo tee "$CONFIG_FILE" > /dev/null << EOF
server:
  http_listen_port: $PROMTAIL_PORT
  grpc_listen_port: 0
  log_level: warn

positions:
  filename: $POSITIONS_FILE

clients:
  - url: http://$MONITORING_SERVER:3100/loki/api/v1/push
    timeout: 10s
    backoff_config:
      min_period: 500ms
      max_period: 5m
      max_retries: 10

scrape_configs:
  # ── JOB 1: 애플리케이션 로그 (멀티라인 + 에러 필터) ──
  - job_name: app-logs
    static_configs:
      - targets:
          - localhost
        labels:
          system_name: "$SYSTEM_NAME"
          instance_role: "$INSTANCE_ROLE"
          host: "$HOST_NAME"
          log_type: "application"
          __path__: "$LOG_PATH"
    pipeline_stages:
      # STAGE 1: 포괄적 멀티라인 처리 (Java 스택트레이스, 다양한 날짜 포맷)
      - multiline:
          firstline: '^(\d{4}[-/\.]\d{2}[-/\.]\d{2}|\[\d{4}[-/\.]\d{2}[-/\.]\d{2}|\d{2}:\d{2}:\d{2}|\[(?i)(error|warn|info|debug|fatal|critical|trace)\]|(?i)(ERROR|WARN|FATAL|CRITICAL)[:\s]|[A-Z][a-z]{2}\s+\d{1,2}\s+\d{2}:\d{2}:\d{2})'
          max_wait_time: 3s
          max_lines: 200
      # STAGE 2: 빈줄 제거
      - drop:
          expression: '^\s*$'
      # STAGE 3: 에러 키워드 필터 (Loki 저장량 절감)
      # → ERROR/WARN/FATAL 등 에러 관련 키워드가 포함된 로그만 전송
      - drop:
          expression: >-
            (?i)^(?!.*(\berror\b|\bwarn\b|\bfatal\b|\bcritical\b|\bexception\b|\bfail(ed|ure)?\b|\btimeout\b|\brefused\b|\bdenied\b|\bcorrupt\b|\bpanic\b|\bdead(lock)?\b|\bstack\s*trace\b|\bat\s+[\w\.]+\())
      # STAGE 4: 레벨 라벨 추출
      - regex:
          expression: '(?i)(?P<level>ERROR|WARN(?:ING)?|FATAL|CRITICAL|INFO|DEBUG)'
      - labels:
          level:

  # ── JOB 2: 시스템 로그 (syslog — 에러만 필터) ──
  - job_name: system-logs
    static_configs:
      - targets:
          - localhost
        labels:
          system_name: "$SYSTEM_NAME"
          instance_role: "$INSTANCE_ROLE"
          host: "$HOST_NAME"
          log_type: "system"
          __path__: /var/log/messages
    pipeline_stages:
      - drop:
          expression: >-
            (?i)^(?!.*(\berror\b|\bwarn\b|\bfatal\b|\bcritical\b|\bfail(ed|ure)?\b|\btimeout\b|\brefused\b|\bdenied\b|\bpanic\b))

  # ── JOB 3: 보안 로그 (전량 수집) ──
  - job_name: security-logs
    static_configs:
      - targets:
          - localhost
        labels:
          system_name: "$SYSTEM_NAME"
          instance_role: "$INSTANCE_ROLE"
          host: "$HOST_NAME"
          log_type: "security"
          __path__: /var/log/secure
EOF

    sudo chown -R promtail:promtail "$AGENT_DIR"

    # Systemd 서비스 등록
    sudo tee "$SERVICE_FILE" > /dev/null << EOF
[Unit]
Description=AOMS Promtail (Log Collector)
Documentation=https://grafana.com/docs/loki/latest/clients/promtail/
After=network.target

[Service]
User=promtail
Group=promtail
Type=simple
ExecStart=$AGENT_DIR/promtail -config.file=$CONFIG_FILE
Restart=on-failure
RestartSec=5s

[Install]
WantedBy=multi-user.target
EOF

    sudo systemctl daemon-reload
    sudo systemctl enable --now promtail

    sleep 2
    sudo systemctl is-active promtail \
        && info "promtail 정상 기동 (포트 ${PROMTAIL_PORT})" \
        || error "promtail 기동 실패"
}

    info "jmx_exporter 설치 완료: $AGENT_DIR"

    warn "  -javaagent:$AGENT_DIR/jmx_prometheus_javaagent-0.20.0.jar=${JMX_PORT}:$AGENT_DIR/jmx-config.yml"
    warn "예) CATALINA_OPTS에 추가 후 Tomcat 재시작 필요"

# ── 설치 실행 ────────────────────────────────────────────────

info "=== 설치 완료 ==="
info "Prometheus에서 다음 타겟 추가 필요:"
echo ""
echo "  - targets: ['$(hostname -I | awk '{print $1}'):${NODE_EXPORTER_PORT}']"
```

```powershell
<#
.SYNOPSIS
    AOMS Windows 에이전트 자동 설치 스크립트 (폐쇄망용)

.DESCRIPTION
    windows_exporter 와 promtail 을 Windows Service 로 등록합니다.
    폐쇄망 환경을 위해 로컬 바이너리를 사용합니다.

.PARAMETER SystemName

[CmdletBinding()]

Write-Info "=== AOMS Windows 에이전트 설치 시작 ==="

# ── windows_exporter 설치 ───────────────────────────────────
function Install-WindowsExporter {
    Write-Info "--- windows_exporter 설치 ---"

    $MsiPath = Join-Path $ScriptDir "windows_exporter-0.25.1-amd64.msi"
    if (-not (Test-Path $MsiPath)) { Write-Err "MSI 파일 없음: $MsiPath" }

    $AgentDir = Join-Path $InstallDir "windows_exporter"
    New-Item -ItemType Directory -Force -Path $AgentDir | Out-Null

    # MSI 설치 (자동, 로그 수집 설정 포함)
    $MsiArgs = @(
        "/i", $MsiPath,
        "/qn",
        "ENABLED_COLLECTORS=cpu,memory,logical_disk,net,os,service,system,process",
        "LISTEN_PORT=9182",
        "EXTRA_FLAGS=--collector.service.services-where=`"State='running'`""
    )
    $result = Start-Process -FilePath "msiexec.exe" -ArgumentList $MsiArgs -Wait -PassThru
    if ($result.ExitCode -ne 0) { Write-Err "windows_exporter MSI 설치 실패 (ExitCode: $($result.ExitCode))" }

    # 방화벽 규칙 추가
    New-NetFirewallRule -DisplayName "AOMS windows_exporter" `
        -Direction Inbound -Protocol TCP -LocalPort 9182 -Action Allow `
        -ErrorAction SilentlyContinue

    # 서비스 시작 확인
    Start-Sleep -Seconds 3
    $svc = Get-Service -Name "windows_exporter" -ErrorAction SilentlyContinue
    if ($svc -and $svc.Status -eq 'Running') {
        Write-Info "windows_exporter 정상 기동 (포트 9182)"
    } else {
        Write-Err "windows_exporter 기동 실패"
    }
}

# ── promtail 설치 ────────────────────────────────────────────
function Install-Promtail {
    Write-Info "--- promtail 설치 ---"

    $AgentDir   = Join-Path $InstallDir "promtail"
    $ZipPath    = Join-Path $ScriptDir "promtail-windows-amd64.exe.zip"
    $BinPath    = Join-Path $AgentDir "promtail.exe"
    $ConfigPath = Join-Path $AgentDir "promtail.yml"
    $PosPath    = Join-Path $AgentDir "positions.yaml"

    if (-not (Test-Path $ZipPath)) { Write-Err "ZIP 파일 없음: $ZipPath" }

    New-Item -ItemType Directory -Force -Path $AgentDir | Out-Null

    # 압축 해제
    Expand-Archive -Path $ZipPath -DestinationPath $AgentDir -Force
    # ZIP 내 파일명이 다를 수 있으므로 rename
    $extracted = Get-ChildItem $AgentDir -Filter "promtail-windows*.exe" | Select-Object -First 1
    if ($extracted) { Rename-Item $extracted.FullName $BinPath -Force }
    if (-not (Test-Path $BinPath)) { Write-Err "promtail.exe를 찾을 수 없음" }

    # Windows 경로 이스케이프 처리 (백슬래시 → 슬래시)
    $LogPathNorm = $LogPath -replace '\\', '/'
    $PosPathNorm = $PosPath -replace '\\', '/'

    # 설정 파일 생성 (방법3+4 조합: 포괄적 멀티라인 + 에러 키워드 필터)
    @"
server:
  http_listen_port: 9080
  grpc_listen_port: 0
  log_level: warn

positions:
  filename: $PosPathNorm

clients:
  - url: http://${MonitoringServer}:3100/loki/api/v1/push
    timeout: 10s
    backoff_config:
      min_period: 500ms
      max_period: 5m
      max_retries: 10

scrape_configs:
  - job_name: app-logs
    static_configs:
      - targets:
          - localhost
        labels:
          system_name: "$SystemName"
          instance_role: "$InstanceRole"
          host: "$HostName"
          log_type: "application"
          __path__: "$LogPathNorm"
    pipeline_stages:
      # STAGE 1: 포괄적 멀티라인 (Java/.NET 스택트레이스, 다양한 날짜 포맷)
      - multiline:
          firstline: '^(\d{4}[-/\.]\d{2}[-/\.]\d{2}|\[\d{4}[-/\.]\d{2}[-/\.]\d{2}|\d{2}:\d{2}:\d{2}|\[(?i)(error|warn|info|debug|fatal|critical|trace)\]|(?i)(ERROR|WARN|FATAL|CRITICAL)[:\s])'
          max_wait_time: 3s
          max_lines: 200
      # STAGE 2: 빈줄 제거
      - drop:
          expression: '^\s*$'
      # STAGE 3: 에러 키워드 필터 (ERROR/WARN/FATAL 등 포함 로그만 전송)
      - drop:
          expression: >-
            (?i)^(?!.*(\berror\b|\bwarn\b|\bfatal\b|\bcritical\b|\bexception\b|\bfail(ed|ure)?\b|\btimeout\b|\brefused\b|\bdenied\b|\bcorrupt\b|\bpanic\b|\bdead(lock)?\b|\bstack\s*trace\b|\bat\s+[\w\.]+\())
      # STAGE 4: 레벨 라벨 추출
      - regex:
          expression: '(?i)(?P<level>ERROR|WARN(?:ING)?|FATAL|CRITICAL|INFO|DEBUG)'
      - labels:
          level:

  - job_name: windows-event-logs
    static_configs:
      - targets:
          - localhost
        labels:
          system_name: "$SystemName"
          instance_role: "$InstanceRole"
          host: "$HostName"
          log_type: "system"
          __path__: "C:/Windows/System32/winevt/Logs/Application.evtx"
    pipeline_stages:
      - drop:
          expression: >-
            (?i)^(?!.*(\berror\b|\bwarn\b|\bfatal\b|\bcritical\b|\bfail(ed|ure)?\b|\btimeout\b))
"@ | Set-Content -Path $ConfigPath -Encoding UTF8

    # Windows Service 등록 (sc.exe 사용)
    $svcName = "aoms-promtail"
    $svcDesc = "AOMS Promtail Log Collector"

    # 기존 서비스 제거 (재설치 시)
    $existing = Get-Service -Name $svcName -ErrorAction SilentlyContinue
    if ($existing) {
        Stop-Service -Name $svcName -Force -ErrorAction SilentlyContinue
        sc.exe delete $svcName | Out-Null
        Start-Sleep -Seconds 2
    }

    # 서비스 등록
    New-Service -Name $svcName `
        -DisplayName $svcDesc `
        -BinaryPathName "`"$BinPath`" -config.file=`"$ConfigPath`"" `
        -StartupType Automatic `
        -Description $svcDesc

    # 방화벽 규칙 추가
    New-NetFirewallRule -DisplayName "AOMS Promtail" `
        -Direction Inbound -Protocol TCP -LocalPort 9080 -Action Allow `
        -ErrorAction SilentlyContinue

    Start-Sleep -Seconds 3
    $svc = Get-Service -Name $svcName -ErrorAction SilentlyContinue
    if ($svc -and $svc.Status -eq 'Running') {
        Write-Info "promtail 정상 기동"
    } else {
        Write-Err "promtail 기동 실패 — 로그 확인: Get-EventLog -LogName Application -Source $svcName"
    }
}

# ── 설치 실행 ────────────────────────────────────────────────
```
