### T4.3 — Loki 3.x API 로그 조회 (핵심 유의사항)

```python
import re
import requests
from datetime import datetime, timedelta, timezone

def _to_loki_ns(dt: datetime) -> str:
    """datetime → Loki 3.x nanosecond 타임스탬프 변환 (정밀도 손실 방지)"""
    return str(int(dt.timestamp()) * 1_000_000_000)

def mask_sensitive_data(text: str) -> str:
    """PII 및 결제정보 마스킹 처리"""
    text = re.sub(r'\b\d{4}[-\s]?\d{4}[-\s]?\d{4}[-\s]?\d{4}\b', '****-****-****-****', text)  # 카드번호
    text = re.sub(r'\b\d{6}[-\s]?\d{7}\b', '******-*******', text)   # 주민등록번호
    text = re.sub(r'\b01[0-9][-\s]?\d{3,4}[-\s]?\d{4}\b', '010-****-****', text)  # 전화번호
    text = re.sub(r'[\w.-]+@[\w.-]+\.\w+', '***@***.***', text)       # 이메일
    return text

def fetch_error_logs():
    """
    최근 5분간 ERROR/WARN/FATAL 로그 조회

    Loki 3.x API 유의사항:
    - /loki/api/v1/query_range (GET만 사용) — POST 시 400 반환
    - start/end: nanosecond 정수 문자열 또는 RFC3339 포맷
    - direction=forward: 시간순 정렬
    """
    now   = datetime.now(timezone.utc)
    start = now - timedelta(minutes=5)

    query = '{log_type="application"} |~ "(?i)(error|warn|fatal|exception|critical)"'

    params = {
        "query":     query,
        "start":     _to_loki_ns(start),
        "end":       _to_loki_ns(now),
        "limit":     "500",
        "direction": "forward",
    }

    for attempt in range(1, 4):   # 최대 3회 재시도
        try:
            resp = requests.get(
                f"{LOKI_URL}/loki/api/v1/query_range",
                params=params,
                timeout=30
            )
            if resp.status_code == 400:
                logging.error(f"Loki 쿼리 오류(400): {resp.text[:300]}")
                return None
            resp.raise_for_status()
            return resp.json()
        except requests.exceptions.Timeout:
            logging.warning(f"Loki 조회 타임아웃 (시도 {attempt}/3)")
        except requests.exceptions.RequestException as e:
            logging.error(f"Loki 조회 실패 (시도 {attempt}/3): {e}")
            if attempt == 3:
                return None
    return None
```

### T4.4 — 담당자별 LLM API key 분리 (AI 비용 분리 청구)

시스템 담당자마다 별도 LLM API key를 사용하여 AI 사용 비용을 분리 청구합니다.

**DB 설계**: `contacts.llm_api_key` 컬럼에 담당자별 API key 저장 (NULL이면 시스템 기본값 사용)

**API key 조회 로직** (Log Analyzer):

```python
def get_api_key_for_system(system_name: str) -> str:
    """
    시스템의 primary 담당자 API key 조회.
    1순위: 담당자(contacts)의 llm_api_key
    2순위: 환경변수 LLM_API_KEY (기본값)

    비용 분리 원칙:
    - 각 담당자가 자신의 API key를 등록하면 해당 키로 LLM 호출
    - API key 미등록 담당자는 공용 키 사용 (관리자 비용)
    """
    resp = requests.get(
        f"{ADMIN_API_URL}/api/v1/systems/{system_name}/contacts",
        timeout=5
    )
    if resp.status_code == 200:
        contacts = resp.json()
        for contact in contacts:
            if contact.get("role") == "primary" and contact.get("llm_api_key"):
                return contact["llm_api_key"]
    return os.environ.get("LLM_API_KEY", "")

def analyze_with_llm(system_name: str, logs: list) -> dict:
    """담당자별 API key로 LLM 호출"""
    api_key = get_api_key_for_system(system_name)

    headers = {
        "Content-Type": "application/json",
        "Authorization": f"Bearer {api_key}"
    }

    payload = {
        "model": os.environ.get("LLM_MODEL_NAME", "default"),
        "messages": [
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user", "content": USER_PROMPT.format(
                system_name=system_name,
                instance_role=logs[0].get("instance_role", ""),
                host=logs[0].get("host", ""),
                count=len(logs),
                log_content=mask_sensitive_data("\n".join(l["line"] for l in logs[:50]))
            )}
        ]
    }

    resp = requests.post(LLM_API_URL, json=payload, headers=headers, timeout=60)
    resp.raise_for_status()
    return resp.json()
```

### T4.8 — 오류 피드백 수집 서버

> **⚠️ Phase 4c 변경사항**: Flask 피드백 서버(포트 8081)는 **n8n Webhook으로 대체**되었습니다.
> - `docker-compose.yml`에서 `feedback-server` 서비스를 제거하세요.
> - 피드백 수신은 n8n 워크플로우 3 (Webhook: `POST /webhook/feedback`)이 처리합니다.
> - 기존 JSONL 데이터는 `phase4c-n8n.md`의 마이그레이션 스크립트로 PostgreSQL에 이전합니다.
> - 상세 내용: [Phase 4c — n8n 자동화](./phase4c-n8n.md)

담당자가 알림 수신 후 해결 내용을 기록하는 웹 서버. 수집된 피드백은 LLM 분석 시 컨텍스트로 활용.

```python
"""
오류 해결 피드백 수집 웹서버 (포트 8081)
담당자가 알림 링크를 통해 접속하여 해결책을 기록
"""
from flask import Flask, request, jsonify, render_template_string
import json, os
from datetime import datetime

app = Flask(__name__)
FEEDBACK_FILE = "/opt/monitoring/data/alert_feedback.jsonl"
os.makedirs(os.path.dirname(FEEDBACK_FILE), exist_ok=True)

FORM_HTML = """<!DOCTYPE html>
<html lang="ko"><head><meta charset="UTF-8"><title>오류 해결 기록</title>
<style>body{font-family:Arial;max-width:600px;margin:40px auto;padding:20px}
label{display:block;margin-top:16px;font-weight:bold}
input,textarea,select{width:100%;padding:8px;margin-top:4px;box-sizing:border-box}
button{margin-top:20px;padding:10px 24px;background:#0078d4;color:#fff;border:none;cursor:pointer}
</style></head><body>
<h2>오류 해결 내용 기록</h2>
<form method="POST" action="/feedback">
  <input type="hidden" name="alert_id" value="{{ alert_id }}">
  <input type="hidden" name="system" value="{{ system }}">
  <label>시스템</label><input type="text" value="{{ system }}" readonly>
  <label>오류 유형</label>
  <select name="error_type">
    <option>DB 연결 오류</option><option>메모리 부족</option>
    <option>디스크 부족</option><option>네트워크 오류</option>
    <option>인증 실패</option><option>타임아웃</option><option>기타</option>
  </select>
  <label>해결 내용 (구체적으로 작성)</label>
  <textarea name="solution" rows="5" required
    placeholder="예: DB 커넥션 풀 최대값을 50→100으로 증가, application.yml 수정 후 재기동"></textarea>
  <label>처리자</label><input type="text" name="resolver" placeholder="홍길동">
  <button type="submit">저장</button>
</form></body></html>"""

@app.route('/feedback-form')
def feedback_form():
    return render_template_string(FORM_HTML,
        alert_id=request.args.get('alert_id', ''),
        system=request.args.get('system', ''))

@app.route('/feedback', methods=['POST'])
def save_feedback():
    data = {
        "alert_id": request.form.get('alert_id'),
        "system": request.form.get('system'),
        "error_type": request.form.get('error_type'),
        "solution": request.form.get('solution'),
        "resolver": request.form.get('resolver'),
        "saved_at": datetime.now().isoformat()
    }
    with open(FEEDBACK_FILE, 'a', encoding='utf-8') as f:
        f.write(json.dumps(data, ensure_ascii=False) + '\n')
    return "<h3>해결 내용이 저장되었습니다.</h3>"

@app.route('/feedback/<system>', methods=['GET'])
def get_feedback(system):
    """특정 시스템의 최근 피드백 조회 (LLM 분석 시 컨텍스트 제공)"""
    results = []
    if os.path.exists(FEEDBACK_FILE):
        with open(FEEDBACK_FILE, encoding='utf-8') as f:
            for line in f:
                item = json.loads(line)
                if item.get('system') == system:
                    results.append(item)
    return jsonify(results[-10:])

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8081)
```
